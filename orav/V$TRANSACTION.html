List the Undo per session

SELECT
s.sid,
s.username,
r.name "RBS name",
tx.start_time,
tx.used_ublk "Undo blocks",
tx.used_urec "Undo recs"
FROM
v$session s,
v$transaction tx,
v$rollname r
WHERE
tx.addr = s.taddr and
r.usn = tx.xidusn;
